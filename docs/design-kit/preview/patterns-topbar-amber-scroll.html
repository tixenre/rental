/* 1. CSS — background declarativo, sin JS por frame */
.topbar {
  background: color-mix(
    in oklch,
    var(--amber) var(--amber-pct, 0%),
    color-mix(in oklch, var(--background) 92%, transparent)
  );
  backdrop-filter: blur(16px);
  transition: background 80ms linear;
}

/* 2. Snap a 65% — invierte elementos internos */
.topbar-snap .tb-dot  { background: var(--background); }
.topbar-snap .tb-date {
  border-color: color-mix(in oklch, var(--background) 80%, transparent);
  background:   color-mix(in oklch, var(--background) 25%, transparent);
}
.topbar-snap .tb-user {
  border-color: transparent;
  background:   color-mix(in oklch, var(--background) 30%, transparent);
}

/* 3. Página: setea --amber-pct (src/routes/index.tsx) */
const heroRef = useRef(null);
useEffect(() => {
  const hero = heroRef.current;
  const onScroll = () => {
    const heroH = hero.offsetHeight;
    const pct = heroH > 0
      ? Math.min(100, Math.round((window.scrollY / heroH) * 100))
      : 0;
    document.documentElement.style.setProperty("--amber-pct", pct + "%");
    setScrolled(pct >= 65); // retinta el cat-bar también
  };
  window.addEventListener("scroll", onScroll, { passive: true });
  onScroll();
  return () => {
    window.removeEventListener("scroll", onScroll);
    document.documentElement.style.setProperty("--amber-pct", "0%");
  };
}, []);

/* 4. TopBar: lee --amber-pct via getComputedStyle (TopBar.tsx) */
const onScroll = () => {
  const pct = parseFloat(
    getComputedStyle(document.documentElement)
      .getPropertyValue("--amber-pct") || "0"
  );
  header.style.background = `color-mix(
    in oklch, var(--amber) ${pct}%,
    color-mix(in oklch, var(--background) 92%, transparent)
  )`;
  setSnapped(pct >= 65); // ← snap point canónico Rambla
};

/* 5. Snap: logo invert, date-pill bone sólido, cart btn ink/amber */
// Logo
<Logo className={cn(snapped && "[filter:brightness(0)_invert(1)]")} />
// Date pill (snapped)
"border-background/80 bg-background text-ink hover:bg-background/90"
// Cart btn (snapped)
"bg-ink text-amber hover:opacity-90"  // inversión completa
// User btn (snapped)
"border-background/80 bg-background text-ink hover:bg-background/90"
Snap point: 65 % es el valor específico de Rambla — no un genérico. Si se cambia, actualizar también el repo (components/topbar.tsx). · Hyperparameter: el 92% en el color-mix de base controla qué tan "bone" arranca el fondo. Si queda muy amarillo a pct=30%, bajarlo a 85–88%. · No aplica en /estudio (hero ink oscuro) — solo en pantallas con hero amber.
