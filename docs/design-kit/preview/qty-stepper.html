/* En 0 NO mostrás stepper — mostrás un botón + que lo crea.
   Bajar a 0 desde el stepper = quitar del carrito (+ toast undo). */
function dec() {
  if (value === 1) removeFromCart(id);   // dispara toast "Deshacer"
  else onChange(value - 1);
}
